        # Database Persistence Section
        st.markdown("---")
        st.subheader("Save Run to Database")
        
        col1, col2 = st.columns(2)
        with col1:
            scenario_id = st.text_input(
                "Scenario ID (UUID)",
                value="00000000-0000-0000-0000-000000000000",
                help="Unique identifier for this scenario"
            )
        with col2:
            preference_set_id = st.text_input(
                "Preference Set ID (UUID)",
                value="00000000-0000-0000-0000-000000000001",
                help="Unique identifier for this preference set"
            )
        
        executed_by = st.text_input(
            "Executed by (optional)",
            value="",
            placeholder="your name or identifier"
        )
        
        if st.button("Save & Run to Database", type="primary", key="save_to_db"):
            if not model.attributes:
                st.error("No attributes defined. Please add attributes in Setup tab.")
            elif not model.alternatives:
                st.error("No alternatives defined. Please add alternatives in Setup tab.")
            else:
                try:
                    engine = get_engine()
                    vft_service = VftService(engine)
                    run_id = vft_service.execute_vft_run(
                        scenario_id=scenario_id,
                        preference_set_id=preference_set_id,
                        model=model,
                        scaling_type="Linear",
                        executed_by=executed_by or "vft_ui"
                    )
                    st.success(f"✓ VFT run saved with ID: `{run_id}`")
                    
                    with st.expander("Verification SQL Queries"):
                        st.code(f"""-- Check VFT run was created
SELECT * FROM runs WHERE run_id = '{run_id}';

-- Check vft_run_config
SELECT * FROM vft_run_config WHERE run_id = '{run_id}';

-- Check vft_criterion_utilities
SELECT * FROM vft_criterion_utilities WHERE run_id = '{run_id}';

-- Check vft_weighted_utilities
SELECT * FROM vft_weighted_utilities WHERE run_id = '{run_id}';

-- Check result_scores
SELECT * FROM result_scores WHERE run_id = '{run_id}';
""", language="sql")
                except Exception as e:
                    st.error(f"Failed to save run: {str(e)}")

## What Happens on Button Click

```
Button Click
    ↓
get_engine() loads .env DATABASE_URL
    ↓
VftService.execute_vft_run() called
    ↓
with engine.begin() as conn:
    ├─ RunRepo.create_run(...) 
    │  └─ INSERT runs → get run_id
    ├─ VftRepo.save_run_config(run_id)
    │  └─ UPSERT vft_run_config
    ├─ VftRepo.replace_criterion_utilities(run_id)
    │  └─ DELETE + INSERT vft_criterion_utilities
    ├─ VftRepo.replace_weighted_utilities(run_id)
    │  └─ DELETE + INSERT vft_weighted_utilities
    └─ VftRepo.replace_result_scores(run_id)
       └─ DELETE + INSERT result_scores
    ↓
COMMIT (or ROLLBACK on exception)
    ↓
UI shows success message with run_id
```
